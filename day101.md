### Easy (5 problems, 90 seconds each)

**Q1.** Find the top 3 customers by total spending.
```sql
-- customers: id, name
-- orders: id, customer_id, total_amount
```

**Q2.** Count orders per status, show only statuses with > 10 orders.

**Q3.** Find orders placed on weekends (Saturday/Sunday).

**Q4.** Show the average order amount per month in 2026.

**Q5.** Find customers whose name starts with 'Lim' (common SG surname).

<details>
<summary>🔑 Answers 1-5</summary>

```sql
-- A1
SELECT c.name, SUM(o.total_amount) as total_spent
FROM customers c JOIN orders o ON c.id = o.customer_id
GROUP BY c.name ORDER BY total_spent DESC LIMIT 3;

-- A2
SELECT status, COUNT(*) FROM orders GROUP BY status HAVING COUNT(*) > 10;

-- A3
SELECT * FROM orders WHERE EXTRACT(ISODOW FROM order_date) IN (6, 7);

-- A4
SELECT DATE_TRUNC('month', order_date) as month, AVG(total_amount)
FROM orders WHERE order_date >= '2026-01-01' GROUP BY month ORDER BY month;

-- A5
SELECT * FROM customers WHERE name LIKE 'Lim%';
```

```sql
-- A16
WITH daily AS (
    SELECT DISTINCT customer_id, DATE(order_date) as dt
),
grouped AS (
    SELECT customer_id, dt,
           dt - (ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY dt))::int as grp
    FROM daily
)
SELECT customer_id, MAX(streak) as longest_streak FROM (
    SELECT customer_id, grp, COUNT(*) as streak FROM grouped GROUP BY customer_id, grp
) s GROUP BY customer_id ORDER BY longest_streak DESC;

-- A17
WITH gaps AS (
    SELECT customer_id, order_date,
           order_date - LAG(order_date) OVER (PARTITION BY customer_id ORDER BY order_date) as gap
    FROM orders
)
SELECT customer_id, ROUND(AVG(gap), 1) as avg_days_between
FROM gaps WHERE gap IS NOT NULL GROUP BY customer_id;

-- A18
WITH monthly_rev AS (
    SELECT restaurant_id, DATE_TRUNC('month', order_date) as m, SUM(total_amount) as rev
    FROM orders GROUP BY 1, 2
),
best AS (
    SELECT restaurant_id, MAX(rev) as best_rev FROM monthly_rev GROUP BY restaurant_id
),
current AS (
    SELECT restaurant_id, SUM(total_amount) as curr_rev
    FROM orders WHERE order_date >= DATE_TRUNC('month', CURRENT_DATE) GROUP BY 1
)
SELECT b.restaurant_id, c.curr_rev, b.best_rev
FROM best b JOIN current c ON b.restaurant_id = c.restaurant_id
WHERE c.curr_rev < b.best_rev * 0.5;

-- A19
SELECT customer_id, COUNT(DISTINCT restaurant_id) as restaurant_count
FROM orders GROUP BY customer_id ORDER BY restaurant_count DESC LIMIT 3;

-- A20
WITH cohorts AS (
    SELECT c.id, DATE_TRUNC('month', c.signup_date) as cohort
    FROM customers c
),
activity AS (
    SELECT DISTINCT o.customer_id, DATE_TRUNC('month', o.order_date) as active_month
    FROM orders o
),
cohort_activity AS (
    SELECT co.id, co.cohort, a.active_month,
           EXTRACT(YEAR FROM a.active_month)*12 + EXTRACT(MONTH FROM a.active_month)
           - EXTRACT(YEAR FROM co.cohort)*12 - EXTRACT(MONTH FROM co.cohort) as month_n
    FROM cohorts co JOIN activity a ON co.id = a.customer_id
)
SELECT cohort,
       COUNT(DISTINCT CASE WHEN month_n = 0 THEN id END) as m0,
       COUNT(DISTINCT CASE WHEN month_n = 1 THEN id END) as m1,
       COUNT(DISTINCT CASE WHEN month_n = 2 THEN id END) as m2,
       COUNT(DISTINCT CASE WHEN month_n = 3 THEN id END) as m3,
       COUNT(DISTINCT id) as total
FROM cohort_activity GROUP BY cohort ORDER BY cohort;
```

**P4.** Write a function that paginates through an API (given a fetch function) and returns all results.

<details>
<summary>🔑 Answer</summary>

```python
def fetch_all_pages(fetch_func, page_size=100):
    all_results = []
    page = 0
    while True:
        response = fetch_func(page=page, size=page_size)
        data = response.get('data', [])
        all_results.extend(data)
        if len(data) < page_size:
            break
        page += 1
    return all_results
```

```python
import pandas as pd

def dict_list_to_df(data, dtype_map=None):
    df = pd.DataFrame(data)
    if dtype_map:
        for col, dtype in dtype_map.items():
            if col in df.columns:
                df[col] = df[col].astype(dtype)
    # Convert date strings
    date_cols = [c for c in df.columns if 'date' in c.lower()]
    for col in date_cols:
        df[col] = pd.to_datetime(df[col], errors='coerce')
    return df
```

**P6.** Write a retry decorator that retries a function up to 3 times on exception.

<details>
<summary>🔑 Answer</summary>

```python
import time
from functools import wraps

def retry(max_retries=3, delay=1):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_retries - 1:
                        raise
                    print(f"Attempt {attempt+1} failed: {e}. Retrying in {delay}s...")
                    time.sleep(delay * (attempt + 1))  # exponential backoff
        return wrapper
    return decorator

@retry(max_retries=3, delay=2)
def fetch_api_data(url):
    import requests
    return requests.get(url, timeout=10).json()
```
</details>

**P7.** Write a function that upserts (insert or update) data into PostgreSQL.

<details>
<summary>🔑 Answer</summary>

```python
import psycopg2

def upsert_orders(conn_params, orders_df, table='orders'):
    conn = psycopg2.connect(**conn_params)
    cur = conn.cursor()
    
    for _, row in orders_df.iterrows():
        cur.execute(f"""
            INSERT INTO {table} (order_id, customer_id, amount, status, order_date)
            VALUES (%s, %s, %s, %s, %s)
            ON CONFLICT (order_id) 
            DO UPDATE SET amount = EXCLUDED.amount, 
                          status = EXCLUDED.status
        """, (row['order_id'], row['customer_id'], row['amount'], 
              row['status'], row['order_date']))
    
    conn.commit()
    cur.close()
    conn.close()
```

**P8.** Write a function that calculates basic data quality stats for a DataFrame.

<details>
<summary>🔑 Answer</summary>

```python
def data_quality_report(df):
    report = {
        'total_rows': len(df),
        'total_columns': len(df.columns),
        'null_counts': df.isnull().sum().to_dict(),
        'duplicate_rows': df.duplicated().sum(),
        'memory_mb': round(df.memory_usage(deep=True).sum() / 1e6, 2)
    }
    for col in df.select_dtypes(include='number').columns:
        report[f'{col}_range'] = (df[col].min(), df[col].max())
    return report
```

```python
import pandas as pd

def etl_order_data(input_csv, output_parquet):
    # Extract
    df = pd.read_csv(input_csv)
    print(f"Extracted {len(df)} rows")
    
    # Clean
    df = df.drop_duplicates(subset=['order_id'])
    df = df.dropna(subset=['order_id', 'customer_id', 'amount'])
    df['amount'] = pd.to_numeric(df['amount'], errors='coerce').fillna(0)
    df['order_date'] = pd.to_datetime(df['order_date'], errors='coerce')
    
    # Transform
    df['amount_sgd'] = df['amount'] * 1.0  # assume already SGD
    df['order_month'] = df['order_date'].dt.to_period('M').astype(str)
    df['is_weekend'] = df['order_date'].dt.dayofweek >= 5
    
    # Load
    df.to_parquet(output_parquet, index=False)
    print(f"Loaded {len(df)} rows to {output_parquet}")
    return df
```
