```sql
-- models/marts/customer_segmentation.sql
{{ config(materialized="table") }}

WITH customer_metrics AS (
    SELECT
        customer_id,
        customer_name,
        city,
        loyalty_tier,
        total_orders,
        total_spent,
        first_order_date,
        last_order_date
    FROM {{ ref('dim_customers') }}
),

segmented AS (
    SELECT
        *,
        -- Recency: days since last order
        CURRENT_DATE - last_order_date AS days_since_last_order,
        -- Frequency: total orders
        total_orders AS frequency,
        -- Monetary: total spent
        total_spent AS monetary,
        -- RFM Score
        CASE
            WHEN total_spent >= 30 AND total_orders >= 3 THEN 'Champions'
            WHEN total_spent >= 30 THEN 'Big Spenders'
            WHEN total_orders >= 3 THEN 'Loyal Customers'
            WHEN days_since_last_order <= 30 THEN 'Recent Customers'
            WHEN total_spent >= 15 THEN 'Promising'
            ELSE 'New Customers'
        END AS rfm_segment
    FROM customer_metrics
)

SELECT * FROM segmented
```

### Task 3: Add Tests to Everything (Afternoon, 1 hour)

Update all schema.yml files:

```yaml
# models/marts/schema.yml
version: 2

models:
  - name: dim_customers
    description: "Customer dimension with aggregated metrics"
    tests:
      - dbt_utils.expression_is_true:
          expression: "total_spent >= 0"
    columns:
      - name: customer_id
        tests: [unique, not_null]
      - name: total_spent
        tests:
          - not_null
          - dbt_utils.accepted_range:
              min_value: 0

  - name: fact_orders_enriched
    tests:
      - dbt_utils.expression_is_true:
          expression: "amount > 0"
    columns:
      - name: order_id
        tests: [not_null]
      - name: customer_id
        tests:
          - not_null
          - relationships:
              to: ref('dim_customers')
              field: customer_id

  - name: customer_segmentation
    columns:
      - name: customer_id
        tests: [unique, not_null]
      - name: rfm_segment
        tests:
          - accepted_values:
              values: ['Champions', 'Big Spenders', 'Loyal Customers', 
                       'Recent Customers', 'Promising', 'New Customers']

  - name: monthly_revenue_trends
    columns:
      - name: order_month_str
        tests: [unique, not_null]
```
