### Step 1: Create the S3 Data Lake

```python
"""
aws_setup/s3_setup.py
Create and configure the MakanExpress S3 data lake.
"""

import boto3
import json

BUCKET = 'makanexpress-data-lake-dev'  # change if name taken
REGION = 'ap-southeast-1'

def setup_s3():
    s3 = boto3.client('s3', region_name=REGION)
    
    # Create bucket
    try:
        s3.create_bucket(
            Bucket=BUCKET,
            CreateBucketConfiguration={'LocationConstraint': REGION}
        )
        print(f"✅ Created bucket: {BUCKET}")
    except Exception as e:
        print(f"Bucket exists or error: {e}")
    
    # Block public access
    s3.put_public_access_block(
        Bucket=BUCKET,
        PublicAccessBlockConfiguration={
            'BlockPublicAcls': True,
            'IgnorePublicAcls': True,
            'BlockPublicPolicy': True,
            'RestrictPublicBuckets': True
        }
    )
    
    # Enable versioning
    s3.put_bucket_versioning(
        Bucket=BUCKET,
        VersioningConfiguration={'Status': 'Enabled'}
    )
    
    # Enable encryption
    s3.put_bucket_encryption(
        Bucket=BUCKET,
        ServerSideEncryptionConfiguration={
            'Rules': [{
                'ApplyServerSideEncryptionByDefault': {'SSEAlgorithm': 'AES256'},
                'BucketKeyEnabled': True
            }]
        }
    )
    
    # Lifecycle policy
    s3.put_bucket_lifecycle_configuration(
        Bucket=BUCKET,
        LifecycleConfiguration={
            'Rules': [
                {
                    'ID': 'RawArchive',
                    'Status': 'Enabled',
                    'Filter': {'Prefix': 'raw/'},
                    'Transitions': [
                        {'Days': 30, 'StorageClass': 'STANDARD_IA'},
                        {'Days': 90, 'StorageClass': 'GLACIER'}
                    ],
                    'Expiration': {'Days': 365}
                },
                {
                    'ID': 'TempCleanup',
                    'Status': 'Enabled',
                    'Filter': {'Prefix': 'temp/'},
                    'Expiration': {'Days': 7}
                }
            ]
        }
    )
    
    # Create zone prefixes
    for zone in ['raw', 'processed', 'analytics', 'temp']:
        for table in ['orders', 'customers', 'restaurants', 'order_items', 'payments']:
            s3.put_object(Bucket=BUCKET, Key=f'{zone}/{table}/')
    
    print("✅ S3 data lake configured: versioning, encryption, lifecycle, zones")

if __name__ == '__main__':
    setup_s3()
```

"""
aws_setup/upload_to_s3.py
Upload MakanExpress sample data to S3 data lake.
"""

import boto3
import pandas as pd
import numpy as np
from io import StringIO, BytesIO
from datetime import datetime, timedelta

BUCKET = 'makanexpress-data-lake-dev'
REGION = 'ap-southeast-1'

def generate_sample_data():

        'city': np.random.choice(['Singapore', 'Kuala Lumpur', 'Penang'], 50),
        'signup_date': pd.date_range('2025-06-01', periods=50, freq='5D')
    })
    
    restaurants = pd.DataFrame({
        'restaurant_id': range(1, 16),
        'name': ['Chicken Rice Paradise', 'Nasi Lemak House', 'Roti Prata Corner',
                  'Sakura Sushi', 'Green Curry Thai', 'Dim Sum Dynasty',
                  'Mee Goreng Express', 'Pizza Roma', 'Char Kway Teow Master',
                  'Bak Kut Teh King', 'Satay Street', 'Laksa Legend',
                  'Tandoori Nights', 'Burger Junction', 'Ice Kacang Corner'],
        'cuisine': ['Chinese', 'Malay', 'Indian', 'Japanese', 'Thai', 'Chinese',
                     'Malay', 'Western', 'Chinese', 'Chinese', 'Malay', 'Chinese',
                     'Indian', 'Western', 'Dessert'],
        'city': ['Singapore', 'Kuala Lumpur', 'Singapore', 'Singapore', 'Kuala Lumpur',
                  'Singapore', 'Kuala Lumpur', 'Singapore', 'Penang', 'Kuala Lumpur',
                  'Singapore', 'Singapore', 'Kuala Lumpur', 'Penang', 'Singapore'],
        'rating': [4.5, 4.2, 4.0, 4.7, 4.3, 4.1, 3.8, 4.4, 4.6, 4.5,
                    4.2, 4.3, 4.0, 3.9, 4.1],
        'is_active': [True]*15
    })
    
    # Generate 500 orders over 30 days
    orders = []
    for i in range(1, 501):
        date = datetime(2026, 6, 1) + timedelta(days=np.random.randint(0, 30))
        orders.append({
            'order_id': i,
            'customer_id': np.random.randint(1, 51),
            'restaurant_id': np.random.randint(1, 16),
            'order_date': date.strftime('%Y-%m-%d %H:%M:%S'),
            'total_amount': round(np.random.uniform(5, 50), 2),
            'status': 'completed' if np.random.random() > 0.07 else 'cancelled',
            'delivery_fee': round(np.random.uniform(1, 6), 2)
        })
    orders = pd.DataFrame(orders)
    
    # Generate order items (2-3 per order)
    items = []
    item_id = 1
    for _, order in orders.iterrows():
        for _ in range(np.random.randint(1, 4)):
            items.append({
                'item_id': item_id,
                'order_id': order['order_id'],
                'menu_item_id': np.random.randint(1, 40),
                'quantity': np.random.randint(1, 4),
                'unit_price': round(np.random.uniform(3, 20), 2),
                'discount': round(np.random.choice([0, 0, 0, 0, 1.0]), 2)
            })
            item_id += 1
    order_items = pd.DataFrame(items)
    
    return customers, restaurants, orders, order_items

def upload_raw_data(s3, dataframes):
    """Upload all raw data as CSV to S3."""
    tables = {
        'customers': dataframes[0],
        'restaurants': dataframes[1],
        'orders': dataframes[2],
        'order_items': dataframes[3]
    }
    
    for table_name, df in tables.items():
        csv_buffer = StringIO()
        df.to_csv(csv_buffer, index=False)
        key = f'raw/{table_name}/{table_name}.csv'
        s3.put_object(Bucket=BUCKET, Key=key, Body=csv_buffer.getvalue())
        print(f"  ✅ Uploaded {table_name}: {len(df)} rows → s3://{BUCKET}/{key}")

def upload_processed_data(s3, orders, order_items):
    """Upload processed data as Parquet, partitioned by date."""
    orders['order_date_parsed'] = pd.to_datetime(orders['order_date'])
    
    for date, group in orders.groupby(orders['order_date_parsed'].dt.date):
        date_str = str(date)
        key = f'processed/orders/dt={date_str}/orders.parquet'
        buffer = BytesIO()
        group.drop(columns=['order_date_parsed']).to_parquet(buffer, index=False)
        s3.put_object(Bucket=BUCKET, Key=key, Body=buffer.getvalue())
    
    print(f"  ✅ Uploaded processed orders: 30 date partitions")

def upload_analytics(s3, orders, restaurants):
    """Upload analytics-ready aggregations."""
    orders['order_date_parsed'] = pd.to_datetime(orders['order_date'])
    
    daily_revenue = orders[orders['status'] == 'completed'].groupby(
        [orders['order_date_parsed'].dt.date, 'restaurant_id']
    ).agg(
        total_orders=('order_id', 'count'),
        total_revenue=('total_amount', 'sum'),
        avg_order_value=('total_amount', 'mean')
    ).reset_index()
    
    daily_revenue.columns = ['date', 'restaurant_id', 'total_orders', 'total_revenue', 'avg_order_value']
    
    buffer = BytesIO()
    daily_revenue.to_parquet(buffer, index=False)
    s3.put_object(Bucket=BUCKET, Key='analytics/daily_revenue/daily_revenue.parquet', Body=buffer.getvalue())
    print(f"  ✅ Uploaded analytics: daily_revenue ({len(daily_revenue)} rows)")

if __name__ == '__main__':
    s3 = boto3.client('s3', region_name=REGION)
    
    print("📊 Generating sample data...")
    data = generate_sample_data()
    
    print("\n☁️ Uploading to S3...")
    upload_raw_data(s3, data)
    upload_processed_data(s3, data[2], data[3])
    upload_analytics(s3, data[2], data[1])
    
    print("\n✅ S3 data lake populated!")

### Step 3: Create IAM Roles and Policies

```python
"""
aws_setup/iam_setup.py
Create IAM roles and policies for MakanExpress.
"""

import boto3
import json

def setup_iam():
    iam = boto3.client('iam')
    
    # ─── Data Engineer Group Policy ───
    de_policy = {
        "Version": "2012-10-17",
        "Statement": [
            {
                "Effect": "Allow",
                "Action": [
                    "s3:GetObject", "s3:PutObject", "s3:ListBucket",
                    "s3:GetBucketLocation"
                ],
                "Resource": [
                    "arn:aws:s3:::makanexpress-data-lake-dev",
                    "arn:aws:s3:::makanexpress-data-lake-dev/*"
                ]
            },
            {
                "Effect": "Allow",
                "Action": [
                    "rds:DescribeDBInstances",
                    "rds:DescribeDBSnapshots"
                ],
                "Resource": "*"
            },
            {
                "Effect": "Allow",
                "Action": [
                    "glue:GetDatabase", "glue:GetTable", "glue:GetTables",
                    "glue:StartJobRun", "glue:GetJobRun", "glue:CreateCrawler",
                    "glue:StartCrawler", "glue:GetCrawler"
                ],
                "Resource": "*"
            },
            {
                "Effect": "Allow",
                "Action": [
                    "athena:StartQueryExecution", "athena:GetQueryExecution",
                    "athena:GetQueryResults", "athena:ListWorkGroups"
                ],
                "Resource": "*"
            }
        ]
    }
    
    # Create group
    try:
        iam.create_group(GroupName='MakanExpressDataEngineers')
    except iam.exceptions.EntityAlreadyExistsException:
        pass
    
    iam.put_group_policy(
        GroupName='MakanExpressDataEngineers',
        PolicyName='MakanExpressDEPolicy',
        PolicyDocument=json.dumps(de_policy)
    )
    print("✅ Created MakanExpressDataEngineers group with policy")
    
    # ─── Glue Role ───
    trust_policy = {
        "Version": "2012-10-17",
        "Statement": [{
            "Effect": "Allow",
            "Principal": {"Service": "glue.amazonaws.com"},
            "Action": "sts:AssumeRole"
        }]
    }
    
    try:
        iam.create_role(
            RoleName='MakanExpressGlueRole',
            AssumeRolePolicyDocument=json.dumps(trust_policy)
        )
    except iam.exceptions.EntityAlreadyExistsException:
        pass
    
    iam.attach_role_policy(
        RoleName='MakanExpressGlueRole',
        PolicyArn='arn:aws:iam::aws:policy/service-role/AWSGlueServiceRole'
    )
    
    iam.put_role_policy(
        RoleName='MakanExpressGlueRole',
        PolicyName='S3DataLakeAccess',
        PolicyDocument=json.dumps(de_policy)
    )
    print("✅ Created MakanExpressGlueRole")
    
    print("\n📋 IAM Summary:")
    print("  Group: MakanExpressDataEngineers")
    print("  Role: MakanExpressGlueRole")
    print("  Permissions: S3 (read/write), RDS (describe), Glue, Athena")

if __name__ == '__main__':
    setup_iam()
```

### Step 4: Create RDS Instance and Migrate

```bash
# Create RDS instance
aws rds create-db-instance \
    --db-instance-identifier makanexpress-db \
    --db-instance-class db.t3.micro \
    --engine postgres \
    --engine-version 15.7 \
    --master-username makanexpress_admin \
    --master-user-password 'MakanExpress2026!' \
    --allocated-storage 20 \
    --storage-type gp2 \
    --db-name makanexpress \
    --publicly-accessible \
    --backup-retention-period 7 \
    --region ap-southeast-1

# Wait for it to be ready (~10 minutes)
echo "Waiting for RDS to be available..."
aws rds wait db-instance-available --db-instance-identifier makanexpress-db

# Get endpoint
ENDPOINT=$(aws rds describe-db-instances \
    --db-instance-identifier makanexpress-db \
    --query "DBInstances[0].Endpoint.Address" \
    --output text)
echo "RDS Endpoint: $ENDPOINT"

# Create security group
SG_ID=$(aws ec2 create-security-group \
    --group-name makanexpress-rds-sg \
    --description "MakanExpress RDS access" \
    --query 'GroupId' --output text 2>/dev/null || \
    aws ec2 describe-security-groups \
        --group-names makanexpress-rds-sg \
        --query "SecurityGroups[0].GroupId" --output text)

MY_IP=$(curl -s https://checkip.amazonaws.com)
aws ec2 authorize-security-group-ingress \
    --group-id $SG_ID \
    --protocol tcp --port 5432 \
    --cidr ${MY_IP}/32 2>/dev/null  # ignore if already exists

echo "Security Group: $SG_ID"

# Migrate local database
echo "Migrating local database to RDS..."
PGPASSWORD=your_local_password pg_dump -h localhost -U postgres makanexpress > /tmp/makanexpress_dump.sql
PGPASSWORD='MakanExpress2026!' psql \
    -h $ENDPOINT \
    -U makanexpress_admin \
    -d makanexpress \
    -f /tmp/makanexpress_dump.sql

echo "✅ Migration complete!"
```

"""
aws_setup/verify_rds.py
Verify the RDS deployment is correct.
"""

import boto3
import psycopg2

def verify_rds():
    # Get RDS info
    rds = boto3.client('rds', region_name='ap-southeast-1')
    db = rds.describe_db_instances(DBInstanceIdentifier='makanexpress-db')['DBInstances'][0]
    
    print(f"📊 RDS Instance: {db['DBInstanceIdentifier']}")
    print(f"   Status: {db['DBInstanceStatus']}")
    print(f"   Engine: {db['Engine']} {db['EngineVersion']}")
    print(f"   Class: {db['DBInstanceClass']}")
    print(f"   Storage: {db['AllocatedStorage']} GB")
    print(f"   Endpoint: {db['Endpoint']['Address']}:{db['Endpoint']['Port']}")
    print(f"   Backup: {db['BackupRetentionPeriod']} days")
    
    # Test connection and verify data
    conn = psycopg2.connect(
        host=db['Endpoint']['Address'],
        port=db['Endpoint']['Port'],
        dbname='makanexpress',
        user='makanexpress_admin',
        password='MakanExpress2026!'
    )
    
    cur = conn.cursor()
    
    # Check tables
    cur.execute("""
        SELECT table_name FROM information_schema.tables 
        WHERE table_schema = 'public' 
        ORDER BY table_name
    """)
    tables = [row[0] for row in cur.fetchall()]
    print(f"\n📋 Tables ({len(tables)}): {', '.join(tables)}")
    
    # Check row counts for key tables
    for table in ['raw_orders', 'raw_customers', 'dim_customer', 'fact_order_items']:
        if table in tables:
            cur.execute(f"SELECT COUNT(*) FROM {table}")
            count = cur.fetchone()[0]
            print(f"   {table}: {count} rows")
    
    # Run a test analytical query
    cur.execute("""
        SELECT dr.city, dr.cuisine_type, COUNT(*) as orders, SUM(fo.net_amount) as revenue
        FROM fact_order_items fo
        JOIN dim_restaurant dr ON fo.restaurant_key = dr.restaurant_key
        JOIN dim_date dd ON fo.date_key = dd.date_key
        WHERE fo.order_status = 'completed'
        GROUP BY dr.city, dr.cuisine_type
        ORDER BY revenue DESC
        LIMIT 5
    """)
    
    print("\n📊 Top 5 Revenue by City/Cuisine:")
    for row in cur.fetchall():
        print(f"   {row[0]} / {row[1]}: {row[2]} orders, ${row[3]:.2f}")
    
    conn.close()
    print("\n✅ RDS verification complete!")

if __name__ == '__main__':
    verify_rds()
